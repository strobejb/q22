typedef struct _Elf32_Ehdr
{
    byte       e_ident[EI_NIDENT];      // Magic number and other info
    [enum(ELF_TYPE)]
    Elf32_Half e_type;                  // Object file type
    [enum(ELF_MACHINE)]
    Elf32_Half e_machine;               // Architecture
    Elf32_Word e_version;               // Object file version
    Elf32_Addr e_entry;                 // Entry point virtual address
    Elf32_Off  e_phoff;                 // Program header table file offset
    Elf32_Off  e_shoff;                 // Section header table file offset
    Elf32_Word e_flags;                 // Processor-specific flags
    Elf32_Half e_ehsize;                // ELF header size in bytes
    Elf32_Half e_phentsize;             // Program header table entry size
    Elf32_Half e_phnum;                 // Program header table entry count
    Elf32_Half e_shentsize;             // Section header table entry size
    Elf32_Half e_shnum;                 // Section header table entry count
    Elf32_Half e_shstrndx;              // Section-header string table index
} Elf32_Ehdr, *PElf32_Ehdr;

typedef struct _ELF
{
    byte ei_mag0;                       // ELF magic 0x7f
    byte ei_mag1;                       // ELF magic 'E'
    byte ei_mag2;                       // ELF magic 'L'
    byte ei_mag3;                       // ELF magic 'F'
    [enum(ELF_CLASS)]
    byte ei_class;                      // 32-bit or 64-bit layout
    [enum(ELF_DATA)]
    byte ei_data;                       // Little or big endian encoding
    byte ei_version;                    // ELF identification version
    byte ei_osabi;                      // Target OS ABI
    byte ei_abiversion;                 // ABI version
    byte ei_pad[7];                     // Padding to EI_NIDENT

    [switch_is(ei_class)]
    union
    {
        [case(ELFCLASS32)] Elf32_EhdrTail header32;
        [case(ELFCLASS64)] Elf64_EhdrTail header64;
    };

    [
        offset(header32.e_phoff),
        size_is(header32.e_phnum)
    ]
    Elf32_Phdr programHeaders32[];

    [
        offset(header64.e_phoff),
        size_is(header64.e_phnum)
    ]
    Elf64_Phdr programHeaders64[];

    [
        offset(header32.e_shoff),
        size_is(header32.e_shnum)
    ]
    Elf32_Shdr sectionHeaders32[];

    [
        offset(header64.e_shoff),
        size_is(header64.e_shnum)
    ]
    Elf64_Shdr sectionHeaders64[];

} ELF, *PELF;

[
    export,
    assoc(".elf", ".so", ".o"),
    offset(0),
    endian(ei_data == 2),
    view("elf.sections")
]
ELF elf;
